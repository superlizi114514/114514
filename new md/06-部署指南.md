## 三、准备工作

### 3.1 注册账号

1. **Cloudflare**
   - 网址：https://dash.cloudflare.com/sign-up
   - 免费套餐即可

2. **阿里云**
   - 网址：https://www.aliyun.com
   - 实名认证（发送短信需要）

3. **GitHub**
   - 用于代码托管和 CI/CD

### 3.2 阿里云短信配置

**步骤 1：开通短信服务**
```
1. 登录阿里云控制台
2. 搜索"短信服务"
3. 点击"立即开通"
4. 完成实名认证
```

**步骤 2：创建签名**
```
1. 短信控制台 → 签名管理 → 添加签名
2. 签名名称：山信黑红榜
3. 签名类型：其他
4. 上传证明材料（可选：网站截图、营业执照等）
5. 等待审核（1-2 工作日）
```

**步骤 3：创建模板**
```
1. 短信控制台 → 模板管理 → 添加模板
2. 模板名称：验证码
3. 模板内容：您的验证码是${code}，10 分钟内有效。
4. 适用场景：身份验证
5. 等待审核（1-2 工作日）
```

**步骤 4：获取 AccessKey**
```
1. 阿里云控制台 → 头像 → AccessKey 管理
2. 创建 AccessKey
3. 记录 AccessKey ID 和 Secret
```

**记录以下信息：**
```env
ALIYUN_ACCESS_KEY_ID=LTAI5t...
ALIYUN_ACCESS_KEY_SECRET=xxxxxxxxxxxx
ALIYUN_SIGN_NAME=山信黑红榜
ALIYUN_TEMPLATE_CODE=SMS_182356789  # 审核通过后获取
```

<!-- src/views/Login.vue -->
<template>
  <div class="login-page">
    <h1>山信黑红榜</h1>
    
    <van-form @submit="onSubmit">

      >
        <template #button>
          <van-button
            size="small"
            type="primary"
            :loading="sending"
            :disabled="countdown > 0"
            @click="sendCode"
          >
            {{ countdown > 0 ? `${countdown}秒` : '获取验证码' }}
          </van-button>
        </template>
      </van-field>
      
      <div style="margin: 16px;">
        <van-button round block type="primary" native-type="submit">
          登录
        </van-button>
      </div>
    </van-form>
  </div>

<script setup>
import { ref } from 'vue'
import { showToast } from 'vant'
import { useRouter } from 'vue-router'

const router = useRouter()
const phone = ref('')
const code = ref('')
const sending = ref(false)
const countdown = ref(0)

const sendCode = async () => {
  if (!phone.value || phone.value.length !== 11) {
    showToast('请填写正确的手机号')
    return
  }
  
  sending.value = true
  try {
    const res = await fetch('https://your-worker.your-subdomain.workers.dev/api/send-code', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ phone: phone.value })
    })
    
    const data = await res.json()
    if (data.success) {
      showToast('验证码已发送')
      startCountdown()
    } else {
      showToast(data.message || '发送失败')
    }
  } catch (e) {
    showToast('发送失败，请稍后重试')
  } finally {
    sending.value = false
  }
}

const startCountdown = () => {
  countdown.value = 60
  const timer = setInterval(() => {
    countdown.value--
    if (countdown.value <= 0) {
      clearInterval(timer)
    }
  }, 1000)
}

const onSubmit = async () => {
  try {
    const res = await fetch('https://your-worker.your-subdomain.workers.dev/api/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        phone: phone.value,
        code: code.value
      })
    })
    
    const data = await res.json()
    if (data.success) {
      localStorage.setItem('token', data.token)
      localStorage.setItem('user', JSON.stringify(data.user))
      router.push('/')
    } else {
      showToast(data.message || '登录失败')
    }
  } catch (e) {
    showToast('登录失败，请稍后重试')
  }
}

### 5.3 编写 API 代码

```typescript
// src/index.ts
import { SignJWT, jwtVerify } from 'jose'

const JWT_SECRET = new TextEncoder().encode('your-secret-key-change-in-production')

interface Env {
  ALIYUN_ACCESS_KEY_ID: string
  ALIYUN_ACCESS_KEY_SECRET: string
  ALIYUN_SIGN_NAME: string
  ALIYUN_TEMPLATE_CODE: string
  DB: D1Database
}

export default {
  async fetch(request: Request, env: Env, ctx: ExecutionContext): Promise<Response> {
    const url = new URL(request.url)
    const path = url.pathname
    
    // CORS 处理
    if (request.method === 'OPTIONS') {
      return new Response(null, {
        headers: {
          'Access-Control-Allow-Origin': '*',
          'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
          'Access-Control-Allow-Headers': 'Content-Type',
        }
      })
    }
    
    // 路由
    if (path === '/api/send-code' && request.method === 'POST') {
      return await handleSendCode(request, env)
    }
    
    if (path === '/api/login' && request.method === 'POST') {
      return await handleLogin(request, env)
    }
    
    if (path === '/api/userinfo' && request.method === 'GET') {
      return await handleUserInfo(request, env)
    }
    
    return new Response('Not Found', { status: 404 })
  }
}

// 发送验证码
async function handleSendCode(request: Request, env: Env) {
  try {
    const { phone } = await request.json()
    
    // 验证手机号格式
    if (!/^1[3-9]\d{9}$/.test(phone)) {
      return Response.json({ success: false, message: '手机号格式错误' })
    }
    
    // 生成 6 位验证码
    const code = Math.random().toString().slice(-6)
    
    // 存入 D1 数据库（10 分钟过期）
    const expireAt = new Date(Date.now() + 10 * 60 * 1000)
    await env.DB.prepare(
      'INSERT INTO sms_codes (phone, code, expire_at) VALUES (?, ?, ?)'
    ).bind(phone, code, expireAt.toISOString()).run()
    
    // 调用阿里云短信 API
    const timestamp = new Date().toISOString().replace(/[-:]/g, '').slice(0, 14)
    const signStr = await signAliyunRequest(
      'POST',
      'dysmsapi.aliyuncs.com',
      {
        PhoneNumbers: phone,
        SignName: env.ALIYUN_SIGN_NAME,
        TemplateCode: env.ALIYUN_TEMPLATE_CODE,
        TemplateParam: JSON.stringify({ code })
      },
      env.ALIYUN_ACCESS_KEY_ID,
      env.ALIYUN_ACCESS_KEY_SECRET,
      timestamp
    )
    
    const smsResponse = await fetch(`https://dysmsapi.aliyuncs.com/?${signStr}`, {
      method: 'POST'
    })
    
    const smsResult = await smsResponse.text()
    
    return Response.json({ success: true, message: '验证码已发送' })
  } catch (e) {
    console.error(e)
    return Response.json({ success: false, message: '发送失败' })
  }
}

// 登录
async function handleLogin(request: Request, env: Env) {
  try {
    const { phone, code } = await request.json()
    
    // 验证验证码
    const result = await env.DB.prepare(
      'SELECT * FROM sms_codes WHERE phone = ? AND code = ? AND expire_at > ? AND is_used = 0'
    ).bind(phone, code, new Date().toISOString()).first()
    
    if (!result) {
      return Response.json({ success: false, message: '验证码错误或已过期' })
    }
    
    // 标记验证码已使用
    await env.DB.prepare(
      'UPDATE sms_codes SET is_used = 1 WHERE id = ?'
    ).bind(result.id).run()
    
    // 查询或创建用户
    let user = await env.DB.prepare(
      'SELECT * FROM users WHERE phone = ?'
    ).bind(phone).first()
    
    if (!user) {
      // 创建新用户
      const userId = Date.now().toString()
      await env.DB.prepare(
        'INSERT INTO users (id, phone, created_at) VALUES (?, ?, ?)'
      ).bind(userId, phone, new Date().toISOString()).run()
      
      user = { id: userId, phone }
    }
    
    // 生成 JWT Token
    const token = await new SignJWT({ userId: user.id })
      .setProtectedHeader({ alg: 'HS256' })
      .setExpirationTime('7d')
      .sign(JWT_SECRET)
    
    return Response.json({
      success: true,
      token,
      user: {
        id: user.id,
        phone: user.phone,
        nickname: user.nickname,
        avatar: user.avatar
      }
    })
  } catch (e) {
    console.error(e)
    return Response.json({ success: false, message: '登录失败' })
  }
}

// 获取用户信息
async function handleUserInfo(request: Request, env: Env) {
  try {
    const authHeader = request.headers.get('Authorization')
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      return Response.json({ success: false, message: '未授权' }, { status: 401 })
    }
    
    const token = authHeader.slice(7)
    const { payload } = await jwtVerify(token, JWT_SECRET)
    
    const user = await env.DB.prepare(
      'SELECT id, phone, nickname, avatar, created_at FROM users WHERE id = ?'
    ).bind(payload.userId).first()
    
    if (!user) {
      return Response.json({ success: false, message: '用户不存在' }, { status: 404 })
    }
    
    return Response.json({ success: true, user })
  } catch (e) {
    return Response.json({ success: false, message: 'Token 无效' }, { status: 401 })
  }
}

// 阿里云签名函数
async function signAliyunRequest(method, host, params, accessKeyId, accessKeySecret, timestamp) {
  const sortedParams = Object.keys(params).sort().reduce((acc, key) => {
    acc[key] = params[key]
    return acc
  }, {})
  
  const queryString = new URLSearchParams({
    AccessKeyId: accessKeyId,
    Action: 'SendSms',
    Format: 'JSON',
    RegionId: 'cn-hangzhou',
    SignatureMethod: 'HMAC-SHA1',
    SignatureNonce: crypto.randomUUID(),
    SignatureVersion: '1.0',
    Timestamp: timestamp,
    Version: '2017-05-25',
    ...sortedParams
  }).toString()
  
  const stringToSign = `${method}&%2F&${encodeURIComponent(queryString)}`
  
  const encoder = new TextEncoder()
  const key = encoder.encode(`${accessKeySecret}&`)
  const message = encoder.encode(stringToSign)
  
  const cryptoKey = await crypto.subtle.importKey(
    'raw',
    key,
    { name: 'HMAC', hash: 'SHA-1' },
    false,
    ['sign']
  )
  
  const signature = await crypto.subtle.sign('HMAC', cryptoKey, message)
  const signatureBase64 = btoa(String.fromCharCode(...new Uint8Array(signature)))
  
  return `${queryString}&Signature=${encodeURIComponent(signatureBase64)}`
}
```

```sql
-- schema.sql
CREATE TABLE IF NOT EXISTS users (
  id TEXT PRIMARY KEY,
  phone TEXT UNIQUE NOT NULL,
  nickname TEXT,
  avatar TEXT,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS sms_codes (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  phone TEXT NOT NULL,
  code TEXT NOT NULL,
  is_used INTEGER DEFAULT 0,
  expire_at DATETIME NOT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS profiles (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id TEXT NOT NULL,
  name TEXT NOT NULL,
  campus TEXT,
  class TEXT,
  is_public INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE IF NOT EXISTS merchants (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL,
  category TEXT NOT NULL,
  address TEXT,
  phone TEXT,
  hours TEXT,
  avg_price REAL,
  cover_image TEXT,
  is_certified INTEGER DEFAULT 0,
  certified_at DATETIME,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS profile_reviews (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  reviewer_id TEXT NOT NULL,
  profile_id INTEGER NOT NULL,
  type TEXT NOT NULL CHECK(type IN ('red', 'black')),
  content TEXT NOT NULL,
  is_anonymous INTEGER DEFAULT 0,
  like_count INTEGER DEFAULT 0,
  report_count INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (reviewer_id) REFERENCES users(id),
  FOREIGN KEY (profile_id) REFERENCES profiles(id)
);

CREATE TABLE IF NOT EXISTS merchant_reviews (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  reviewer_id TEXT NOT NULL,
  merchant_id INTEGER NOT NULL,
  rating INTEGER CHECK(rating >= 1 AND rating <= 5),
  type TEXT NOT NULL CHECK(type IN ('red', 'black')),
  content TEXT NOT NULL,
  tags TEXT,
  images TEXT,
  is_anonymous INTEGER DEFAULT 0,
  like_count INTEGER DEFAULT 0,
  report_count INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (reviewer_id) REFERENCES users(id),
  FOREIGN KEY (merchant_id) REFERENCES merchants(id)
);

-- 索引
CREATE INDEX idx_sms_codes_phone ON sms_codes(phone, created_at);
CREATE INDEX idx_profiles_user ON profiles(user_id);
CREATE INDEX idx_profile_reviews_profile ON profile_reviews(profile_id, created_at);
CREATE INDEX idx_merchant_reviews_merchant ON merchant_reviews(merchant_id, created_at);
```

## 十、常见问题

### Q1: 短信发送失败？

**检查：**
1. 阿里云账号是否实名认证
2. 签名和模板是否审核通过
3. AccessKey 是否正确
4. 手机号格式是否正确

### Q2: Workers 请求超时？

**解决：**
- Worker 有 10 秒超时限制
- 复杂操作建议使用异步处理
- 或改用阿里云函数计算

### Q3: 国内访问慢？

**解决：**
