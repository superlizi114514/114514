## 二、赞助流程（半自动方案）

### 2.1 用户端流程

```
1. 用户进入"支持我"页面
2. 查看赞助等级和权益
3. 扫码添加微信
4. 微信转账
5. 返回页面填写赞助信息

7. 等待管理员确认（24 小时内）
8. VIP 开通，收到通知
```

### 2.2 管理员流程

```
1. 微信收到转账
2. 登录管理后台
3. 查看待审核赞助列表
4. 核对手机号/金额
5. 点击"开通 VIP"
6. 选择 VIP 时长（自动推荐）
7. 确认开通
8. 用户自动加入赞助者名单
```

## 三、数据库设计

### 3.1 赞助记录表

```sql
CREATE TABLE support_records (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id TEXT NOT NULL,
  amount REAL NOT NULL,
  status TEXT DEFAULT 'pending' CHECK(status IN ('pending', 'approved', 'rejected')),
  wechat_contact TEXT,
  note TEXT,
  approved_by TEXT,
  approved_at DATETIME,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE INDEX idx_support_records_status ON support_records(status, created_at);
```

### 3.2 赞助者展示表

```sql
CREATE TABLE supporters (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id TEXT,
  nickname TEXT,
  amount REAL,
  message TEXT,
  is_public INTEGER DEFAULT 1,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE INDEX idx_supporters_public ON supporters(is_public, created_at);
```

### 3.3 用户表新增字段

```sql
ALTER TABLE users ADD COLUMN is_vip INTEGER DEFAULT 0;
ALTER TABLE users ADD COLUMN vip_expire DATETIME;
```

## 四、API 设计

### 4.1 提交赞助申请

```typescript
// POST /api/support/submit
// 需要登录

async function handleSubmitSupport(request: Request, env: Env) {
  const { amount, message, wechat } = await request.json()
  const userId = await getUserIdFromToken(request)
  
  // 验证金额
  if (!amount || amount < 1) {
    return Response.json({ 
      success: false, 
      message: '赞助金额至少 ¥1' 
    })
  }
  
  // 创建记录
  await env.DB.prepare(`
    INSERT INTO support_records (user_id, amount, wechat_contact, note, status)
    VALUES (?, ?, ?, ?, 'pending')
  `).bind(userId, amount, wechat, message).run()
  
  return Response.json({ 
    success: true, 
    message: '提交成功！管理员确认后开通 VIP' 
  })
}
```

### 4.2 获取赞助记录

```typescript
// GET /api/support/records

async function handleGetSupportRecords(request: Request, env: Env) {
  const userId = await getUserIdFromToken(request)

    SELECT * FROM support_records WHERE user_id = ? ORDER BY created_at DESC
  `).bind(userId).all()
```

### 4.3 获取赞助者名单

```typescript
// GET /api/supporters
// 公开接口

async function handleGetSupporters(request: Request, env: Env) {
  const supporters = await env.DB.prepare(`
    SELECT s.nickname, s.amount, s.message, s.created_at, u.avatar
    FROM supporters s
    LEFT JOIN users u ON s.user_id = u.id
    WHERE s.is_public = 1
    ORDER BY s.amount DESC, s.created_at DESC
    LIMIT 100
  `).all()
  
  return Response.json({ success: true, supporters })
}
```

### 4.4 管理员审核（后台接口）

```typescript
// POST /api/admin/support/approve
// 需要管理员权限

async function handleApproveSupport(request: Request, env: Env) {
  const { recordId, vipDays } = await request.json()
  
  // 获取记录
  const record = await env.DB.prepare(
    'SELECT * FROM support_records WHERE id = ?'
  ).bind(recordId).first()
  
  if (!record) {
    return Response.json({ success: false, message: '记录不存在' })
  }
  
  // 计算 VIP 过期时间
  const vipExpire = new Date(Date.now() + vipDays * 24 * 60 * 60 * 1000)
  
  // 开通 VIP
  await env.DB.prepare(`
    UPDATE users SET is_vip = 1, vip_expire = ? WHERE id = ?
  `).bind(vipExpire.toISOString(), record.user_id).run()
  
  // 更新记录状态
  await env.DB.prepare(`
    UPDATE support_records 
    SET status = 'approved', approved_by = ?, approved_at = ? 
    WHERE id = ?
  `).bind('admin', new Date().toISOString(), recordId).run()
  
  // 添加到赞助者名单
  await env.DB.prepare(`
    INSERT INTO supporters (user_id, amount, message, is_public)
    SELECT user_id, amount, note, 1 FROM support_records WHERE id = ?
  `).bind(recordId).run()
  
  return Response.json({ success: true })
}
```

### 4.5 管理员拒绝（后台接口）

```typescript
// POST /api/admin/support/reject
// 需要管理员权限

async function handleRejectSupport(request: Request, env: Env) {
  const { recordId, reason } = await request.json()
  
  await env.DB.prepare(`
    UPDATE support_records 
    SET status = 'rejected', note = ? 
    WHERE id = ?
  `).bind(reason, recordId).run()
  
  return Response.json({ success: true })
}
```

### 6.1 管理员表

```sql
CREATE TABLE admins (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  username TEXT UNIQUE NOT NULL,
  password_hash TEXT NOT NULL,
  role TEXT DEFAULT 'admin' CHECK(role IN ('admin', 'super_admin')),
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

### 6.2 管理员登录 API

```typescript
// POST /api/admin/login

async function handleAdminLogin(request: Request, env: Env) {
  const { username, password } = await request.json()
  
  const admin = await env.DB.prepare(
    'SELECT * FROM admins WHERE username = ?'
  ).bind(username).first()
  
  if (!admin) {
    return Response.json({ success: false, message: '用户名或密码错误' })
  }
  
  // 验证密码（使用 bcrypt 或类似库）
  const valid = await verifyPassword(password, admin.password_hash)
  if (!valid) {
    return Response.json({ success: false, message: '用户名或密码错误' })
  }
  
  // 生成管理员 Token
  const token = await new SignJWT({ adminId: admin.id, role: admin.role })
    .setProtectedHeader({ alg: 'HS256' })
    .setExpirationTime('30d')
    .sign(ADMIN_JWT_SECRET)
  
  return Response.json({ success: true, token })
}
```

### 7.2 点评次数检查

```typescript
// 中间件：检查每日点评次数

async function checkReviewLimit(request: Request, env: Env) {
  const userId = await getUserIdFromToken(request)
  const today = new Date().toISOString().slice(0, 10)
  
  // 查询今日已点评次数
  const result = await env.DB.prepare(`
    SELECT COUNT(*) as count FROM profile_reviews 
    WHERE reviewer_id = ? AND DATE(created_at) = ?
  `).bind(userId, today).first()
  
  // 获取用户 VIP 状态
  const user = await env.DB.prepare(
    'SELECT is_vip, vip_expire FROM users WHERE id = ?'
  ).bind(userId).first()
  
  const isVip = user?.is_vip && new Date(user.vip_expire) > new Date()
  const limit = isVip ? 10 : 3  // VIP 10 次，普通 3 次
  
  if (result.count >= limit) {
    return Response.json({ 
      success: false, 
      message: isVip 
        ? '今日 VIP 点评次数已用完（10 次）' 
        : '今日点评次数已用完（3 次），赞助 ¥10 开通 VIP 可享每日 10 次！' 
    }, { status: 403 })
  }
}
```
